react-todo은 구글 로그인을 사용하고, 테이블은 todos 테이블만 사용해

create table public.todos (
  id bigint generated by default as identity not null,
  user_id uuid not null,
  task text null,
  is_completed boolean not null default false,
  inserted_at timestamp with time zone not null default timezone ('utc'::text, now()),
  created_at timestamp with time zone not null default now(),
  constraint todos_pkey primary key (id),
  constraint todos_user_id_fkey foreign KEY (user_id) references auth.users (id),
  constraint todos_task_check check ((char_length(task) > 0))
) TABLESPACE pg_default;

schema visualizer 에서 보면 todos.user_id 는 auth.users.id를 참조하는 외래키로 설정되어 있는거 같은데,
어떻게 처리되는 건지는 잘 모르겠어. 필요하면 users 테이블도 관리 할 수 있도록 해줘.
